//HERE WE WILL COLLECT OBJECT FROM THE SERVER


var assert = require('assert');
module.exports = Response;
/* Test */
describe('Test for Get All Catagories', function () {

	it('should return true if success', function(){
		var isStatus = Response.status;
		var isMessage = Response.message;

		assert.equal(isStatus,true);
		assert.equal(isMessage,"Fetch Categories");
	}); 

	it('should return false if fail', function(){
		var isStatus = Response.status;
		var isMessage = Response.message;

		assert.equal(isStatus,false);
		assert.notEqual(isMessage,"Fetch Categories");
	});

});
