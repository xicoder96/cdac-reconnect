//HERE WE WILL COLLECT OBJECT FROM THE SERVER


var assert = require('assert');
module.exports = Response;
/* Test */
describe('Test for Successful attempt', function () {

	it('should return true if successful attempt', function(){
		var isStatus = Response.status;
		var isMessage = Response.message;
		var isData = Response.data;

		assert.equal(isStatus,true);
		assert.equal(isMessage,"Posted Successfully");
		assert.equal(isData,null);
	}); 

	it('should return false if successful attempt', function(){
		var isStatus = Response.status;
		var isMessage = Response.message;
		var isData = Response.data;

		assert.equal(isStatus,false);
		assert.notEqual(isMessage,"Posted Successfully");
		assert.notEqual(isData,null);
	});

});
